291 static redshow_result_t trace_analyze(Kernel &kernel, uint64_t host_op_id,
292  gpu_patch_buffer_t *trace_data) {
293  redshow_result_t result = REDSHOW_SUCCESS;
294 
295  auto cubin_id = kernel.cubin_id;
296  auto &read_spatial_trace = kernel.read_spatial_trace;
297  auto &write_spatial_trace = kernel.write_spatial_trace;
298  auto &read_temporal_trace = kernel.read_temporal_trace;
299  auto &read_pc_pairs = kernel.read_pc_pairs;
300  auto &write_temporal_trace = kernel.write_temporal_trace;
301  auto &write_pc_pairs = kernel.write_pc_pairs;
302  auto &read_pc_count = kernel.read_pc_count;
303  auto &write_pc_count = kernel.write_pc_count;
304  std::vector<instruction::Symbol> *symbols = NULL;
305  instruction::InstructionGraph *inst_graph = NULL;
306  // Cubin path is added just for debugging purpose
307  std::string cubin_path;
308 
309  cubin_map_lock.lock();
310  if (cubin_map.find(cubin_id) == cubin_map.end()) {
311  result = REDSHOW_ERROR_NOT_EXIST_ENTRY;
312  } else {
313  symbols = &(cubin_map[cubin_id].symbols);
314  inst_graph = &(cubin_map[cubin_id].inst_graph);
315  cubin_path = cubin_map[cubin_id].path;
316  }
317  cubin_map_lock.unlock();
318 
319  // Cubin not found, maybe in the cache map
320  if (result == REDSHOW_ERROR_NOT_EXIST_ENTRY) {
321  uint32_t nsymbols;
322  uint64_t *symbol_pcs;
323  const char *path;
324 
325  cubin_cache_map_lock.lock();
326  if (cubin_cache_map.find(cubin_id) == cubin_cache_map.end()) {
327  result = REDSHOW_ERROR_NOT_EXIST_ENTRY;
328  } else {
329  result = REDSHOW_SUCCESS;
330  auto &cubin_cache = cubin_cache_map[cubin_id];
331  nsymbols = cubin_cache.nsymbols;
332  symbol_pcs = cubin_cache.symbol_pcs;
333  path = cubin_cache.path.c_str();
334  }
335  cubin_cache_map_lock.unlock();
336 
337  if (result == REDSHOW_SUCCESS) {
338  result = redshow_cubin_register(cubin_id, nsymbols, symbol_pcs, path);
339  }
340 
341  // Try fetch cubin again
342  if (result == REDSHOW_SUCCESS) {
343  cubin_map_lock.lock();
344  if (cubin_map.find(cubin_id) == cubin_map.end()) {
345  result = REDSHOW_ERROR_NOT_EXIST_ENTRY;
346  } else {
347  symbols = &(cubin_map[cubin_id].symbols);
348  inst_graph = &(cubin_map[cubin_id].inst_graph);
349  cubin_path = cubin_map[cubin_id].path;
350  }
351  cubin_map_lock.unlock();
352  }
353  }
354 
355  if (result != REDSHOW_SUCCESS) {
356  return result;
357  }
358 
359  MemoryMap *memory_map = NULL;
360 
361  memory_snapshot_lock.lock();
362  auto snapshot_iter = memory_snapshot.upper_bound(host_op_id);
363  if (snapshot_iter == memory_snapshot.begin()) {
364  result = REDSHOW_ERROR_NOT_EXIST_ENTRY;
365  } else {
366  --snapshot_iter;
367  memory_map = &(snapshot_iter->second);
368  }
369  memory_snapshot_lock.unlock();
370 
371  // Memory snapshot not found
372  if (result != REDSHOW_SUCCESS) {
373  return result;
374  }
375 
376  size_t size = trace_data->head_index;
377  gpu_patch_record_t *records = reinterpret_cast<gpu_patch_record_t *>(trace_data->records);
378 
379  for (size_t i = 0; i < size; ++i) {
380  // Iterate over each record
381  gpu_patch_record_t *record = records + i;
382 
383  if (record->size == 0) {
384  // Fast path, no thread active
385  continue;
386  }
387 
388  if (record->flags & GPU_PATCH_BLOCK_ENTER_FLAG) {
389  // Skip analysis
390  } else if (record->flags & GPU_PATCH_BLOCK_EXIT_FLAG) {
391  // Remove temporal records
392  if (analysis_enabled.find(REDSHOW_ANALYSIS_TEMPORAL_REDUNDANCY) != analysis_enabled.end()) {
393  for (size_t j = 0; j < GPU_PATCH_WARP_SIZE; ++j) {
394  if (record->active & (0x1u << j)) {
395  uint32_t flat_thread_id =
396  record->flat_thread_id / GPU_PATCH_WARP_SIZE * GPU_PATCH_WARP_SIZE + j;
397  ThreadId thread_id{record->flat_block_id, flat_thread_id};
398  read_temporal_trace.erase(thread_id);
399  write_temporal_trace.erase(thread_id);
400  }
401  }
402  }
403  } else {
404  uint32_t function_index = 0;
405  uint64_t cubin_offset = 0;
406  uint64_t pc_offset = 0;
407  transform_pc(*symbols, record->pc, function_index, cubin_offset, pc_offset);
408 
409  // record->size * 8, byte to bits
410  instruction::AccessKind access_kind;
411 
412  if (inst_graph->size() != 0) {
413  // Accurate mode, when we have instruction information
414  auto &inst = inst_graph->node(cubin_offset);
415  if (inst.access_kind.get() != NULL) {
416  access_kind = *inst.access_kind;
417  }
418  // Fall back to default mode if failed
419  }
420 
421  if (access_kind.data_type == REDSHOW_DATA_UNKNOWN) {
422  // Default mode, we identify every data as 32 bits unit size, 32 bits vec size, float type
423  access_kind.data_type = default_data_type;
424  access_kind.vec_size = record->size * 8;
425  access_kind.unit_size = MIN2(GPU_PATCH_WARP_SIZE, access_kind.vec_size * 8);
426  }
427 
429  // std::cout << "function_index: " << function_index << ", pc_offset: " <<
430  // pc_offset << ", " << access_kind.to_string() << std::endl;
431  // TODO: accelerate by handling all threads in a warp together
432  for (size_t j = 0; j < GPU_PATCH_WARP_SIZE; ++j) {
433  if ((record->active & (0x1u << j)) == 0) {
434  continue;
435  }
436 
437  uint32_t flat_thread_id =
438  record->flat_thread_id / GPU_PATCH_WARP_SIZE * GPU_PATCH_WARP_SIZE + j;
439  ThreadId thread_id{record->flat_block_id, flat_thread_id};
440 
441  MemoryRange memory_range(record->address[j], record->address[j]);
442  auto iter = memory_map->upper_bound(memory_range);
443  uint64_t memory_op_id = 0;
444  if (iter != memory_map->begin()) {
445  --iter;
446  memory_op_id = iter->second.memory_op_id;
447  }
448 
449  uint32_t address_offset = GLOBAL_MEMORY_OFFSET;
450  if (memory_op_id == 0) {
451  // XXX(Keren): memory_op_id == 1 ?
452  // Memory object not found, it means the memory is local, shared, or allocated in an
453  // unknown way
454  if (record->flags & GPU_PATCH_LOCAL) {
455  memory_op_id = REDSHOW_MEMORY_SHARED;
456  address_offset = LOCAL_MEMORY_OFFSET;
457  } else if (record->flags & GPU_PATCH_SHARED) {
458  memory_op_id = REDSHOW_MEMORY_LOCAL;
459  address_offset = SHARED_MEMORY_OFFSET;
460  } else {
461  // Unknown allocation
462  }
463  }
464 
465  if (memory_op_id == 0) {
466  // Unknown memory object
467  continue;
468  }
469 
470  auto num_units = access_kind.vec_size / access_kind.unit_size;
471  instruction::AccessKind unit_access_kind = access_kind;
472  // We iterate through all the units such that every unit's vec_size = unit_size
473  unit_access_kind.vec_size = unit_access_kind.unit_size;
474 
475  if (record->flags & GPU_PATCH_READ) {
476  read_pc_count[record->pc] += num_units;
477  } else {
478  write_pc_count[record->pc] += num_units;
479  }
480 
481  for (size_t m = 0; m < num_units; m++) {
482  uint64_t value = 0;
483  uint32_t byte_size = unit_access_kind.unit_size >> 3u;
484  memcpy(&value, &record->value[j][m * byte_size], byte_size);
485  value =
486  unit_access_kind.value_to_basic_type(value, decimal_degree_f32, decimal_degree_f64);
487 
488  for (auto analysis : analysis_enabled) {
489  if (analysis == REDSHOW_ANALYSIS_SPATIAL_REDUNDANCY) {
490  if (record->flags & GPU_PATCH_READ) {
491  redundancy::update_spatial_trace(record->pc, value, memory_op_id, unit_access_kind,
492  read_spatial_trace);
493  } else {
494  redundancy::update_spatial_trace(record->pc, value, memory_op_id, unit_access_kind,
495  write_spatial_trace);
496  }
497  } else if (analysis == REDSHOW_ANALYSIS_TEMPORAL_REDUNDANCY) {
498  if (record->flags & GPU_PATCH_READ) {
499  redundancy::update_temporal_trace(
500  record->pc, thread_id, record->address[j] + m * address_offset, value,
501  unit_access_kind, read_temporal_trace, read_pc_pairs);
502  } else {
503  redundancy::update_temporal_trace(
504  record->pc, thread_id, record->address[j] + m * address_offset, value,
505  unit_access_kind, write_temporal_trace, write_pc_pairs);
506  }
507  } else if (analysis == REDSHOW_ANALYSIS_VALUE_FLOW) {
508  } else {
509  // Pass
510  }
511  }
512  }
513  }
514  }
515  }
516 
517  return result;
518 }
