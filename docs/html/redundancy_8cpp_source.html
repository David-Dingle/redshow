15 void update_temporal_trace(u64 pc, ThreadId tid, u64 addr, u64 value,
16  instruction::AccessKind access_kind, TemporalTrace &temporal_trace,
17  PCPairs &pc_pairs) {
18  auto tmr_it = temporal_trace.find(tid);
19  // Record current operation.
20  std::map<u64, std::pair<u64, u64>> record;
21  record[addr] = std::make_pair(pc, value);
22  if (tmr_it == temporal_trace.end()) {
23  // The trace doesn't have the thread's record
24  temporal_trace[tid] = record;
25  } else {
26  // The trace has the thread's record
27  auto m_it = tmr_it->second.find(addr);
28  // m_it: {addr: <pc, value>}
29  if (m_it == tmr_it->second.end()) {
30  // The trace's thread record doesn't have the current addr record.
31  tmr_it->second[addr] = record[addr];
32  } else {
33  auto prev_pc = m_it->second.first;
34  auto prev_value = m_it->second.second;
35  if (prev_value == value) {
36  pc_pairs[pc][prev_pc][std::make_pair(prev_value, access_kind)] += 1;
37  }
38  m_it->second = record[addr];
39  }
40  }
41 }

43 void record_temporal_trace(PCPairs &pc_pairs, PCAccessCount &pc_access_count, u32 pc_views_limit,
44  u32 mem_views_limit, redshow_record_data_t &record_data,
45  TemporalStatistics &temporal_stats, u64 &kernel_temporal_count) {

61  auto val = val_iter.first.first;

63  auto count = val_iter.second;

65  }
66  }
67

69 
70  if (top_views.size() < pc_views_limit) {
71  top_views.push(view);
72  } else {
73  auto &top = top_views.top();
74  if (top.red_count < view.red_count) {
75  top_views.pop();
76  top_views.push(view);
77  }
78  }
79  }
80 
81  auto num_views = 0;
82  // Put top record data views into record_data

85

87  TopRealPCPairs top_real_pc_pairs;

91  for (auto &val_iter : from_pc_iter.second) {
92  auto val = val_iter.first.first;

98  top_real_pc_pairs.push(real_pc_pair);
99  } else {
100  auto &top = top_real_pc_pairs.top();
101  if (top.red_count < real_pc_pair.red_count) {
102  top_real_pc_pairs.pop();
103  top_real_pc_pairs.push(real_pc_pair);
104  }
105  }
106  }
107  }
108 
109  while (top_real_pc_pairs.empty() == false) {
110  auto real_pc_pair = top_real_pc_pairs.top();

114  }

117  record_data.views[num_views].red_count = view.red_count;
118  record_data.views[num_views].access_count = view.access_count;

121  }
