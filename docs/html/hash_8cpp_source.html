47 #define SHA2_SHFR(x, n) (x >> n)
48 #define SHA2_ROTR(x, n) ((x >> n) | (x << ((sizeof(x) << 3) - n)))
49 #define SHA2_ROTL(x, n) ((x << n) | (x >> ((sizeof(x) << 3) - n)))
50 #define SHA2_CH(x, y, z) ((x & y) ^ (~x & z))
51 #define SHA2_MAJ(x, y, z) ((x & y) ^ (x & z) ^ (y & z))
52 #define SHA256_F1(x) (SHA2_ROTR(x, 2) ^ SHA2_ROTR(x, 13) ^ SHA2_ROTR(x, 22))
53 #define SHA256_F2(x) (SHA2_ROTR(x, 6) ^ SHA2_ROTR(x, 11) ^ SHA2_ROTR(x, 25))
54 #define SHA256_F3(x) (SHA2_ROTR(x, 7) ^ SHA2_ROTR(x, 18) ^ SHA2_SHFR(x, 3))
55 #define SHA256_F4(x) (SHA2_ROTR(x, 17) ^ SHA2_ROTR(x, 19) ^ SHA2_SHFR(x, 10))
56 #define SHA2_UNPACK32(x, str) \
57  { \
58  *((str) + 3) = (uint8)((x)); \
59  *((str) + 2) = (uint8)((x) >> 8); \
60  *((str) + 1) = (uint8)((x) >> 16); \

82  uint32 w[64];
83  uint32 wv[8];
84  uint32 t1, t2;
85  const unsigned char *sub_block;
86  int i;
87  int j;
88  for (i = 0; i < (int)block_nb; i++) {
89  sub_block = message + (i << 6);
90  for (j = 0; j < 16; j++) {
91  SHA2_PACK32(&sub_block[j << 2], &w[j]);
92  }
93  for (j = 16; j < 64; j++) {
94  w[j] = SHA256_F4(w[j - 2]) + w[j - 7] + SHA256_F3(w[j - 15]) + w[j - 16];
95  }
96  for (j = 0; j < 8; j++) {
97  wv[j] = m_h[j];
98  }
99  for (j = 0; j < 64; j++) {
100  t1 = wv[7] + SHA256_F2(wv[4]) + SHA2_CH(wv[4], wv[5], wv[6]) + sha256_k[j] + w[j];
101  t2 = SHA256_F1(wv[0]) + SHA2_MAJ(wv[0], wv[1], wv[2]);
102  wv[7] = wv[6];
103  wv[6] = wv[5];
104  wv[5] = wv[4];

106  wv[3] = wv[2];
107  wv[2] = wv[1];
108  wv[1] = wv[0];
109  wv[0] = t1 + t2;
110  }

130 void SHA256::update(const unsigned char *message, unsigned int len) {
131  unsigned int block_nb;
132  unsigned int new_len, rem_len, tmp_len;
133  const unsigned char *shifted_message;
134  tmp_len = SHA224_256_BLOCK_SIZE - m_len;
135  rem_len = len < tmp_len ? len : tmp_len;
136  memcpy(&m_block[m_len], message, rem_len);
137  if (m_len + len < SHA224_256_BLOCK_SIZE) {
138  m_len += len;
139  return;
140  }
141  new_len = len - rem_len;
142  block_nb = new_len / SHA224_256_BLOCK_SIZE;
143  shifted_message = message + rem_len;
144  transform(m_block, 1);
145  transform(shifted_message, block_nb);
146  rem_len = new_len % SHA224_256_BLOCK_SIZE;
147  memcpy(m_block, &shifted_message[block_nb << 6], rem_len);
148  m_len = rem_len;
149  m_tot_len += (block_nb + 1) << 6;
150 }
